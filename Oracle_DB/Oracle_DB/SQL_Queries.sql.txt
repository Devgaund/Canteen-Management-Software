/**** USERS_DETAILS_TABLE	
 	CREATE TABLE USERS_DETAILS(
 		FIRST_NAME VARCHAR2(20) CONSTRAINT NN_FIRST_NAME NOT NULL,
 		LAST_NAME VARCHAR2(20) CONSTRAINT NN_LAST_NAME NOT NULL,
 		DEPT_NAME CHAR(10) CONSTRAINT NN_DEPT_NAME NOT NULL,
 		COURSE_NAME VARCHAR2(20) CONSTRAINT NN_COURSE_NAME NOT NULL,
 		EMAIL_ID VARCHAR2(50) CONSTRAINT NN_EMAIL_ID NOT NULL,
 		MOB_NO NUMBER(11) CONSTRAINT NN_MOB_NO NOT NULL,
 		GR_NO CHAR(20) CONSTRAINT PK_GR_NO PRIMARY KEY,
		GENDER CHAR(6) CONSTRAINT NN_GENDER NOT NULL
 	);
 ****/

/**** LOGIN_DETAILS TABLE
	CREATE TABLE LOGIN_DETAILS(
		GR_NO CHAR(20), 
		PASSWORD VARCHAR2(30) CONSTRAINT NN_PASSWORD NOT NULL,
		CONF_PASSWORD VARCHAR2(30) CONSTRAINT NN_CONF_PASSWORD NOT NULL,
CONSTRAINT FK_LOGIN_DETAILS_GRNO FOREIGN KEY(GR_NO) REFERENCES USERS_DETAILS(GR_NO) ON DELETE CASCADE
	); 
 ****/ 

/**** ORDERS_TABLE	
 	CREATE TABLE ORDERS(
 		ORDER_ID NUMBER(8) CONSTRAINT NN_ORDER_ID NOT NULL UNIQUE,
 		DEPT_NAME CHAR(10) CONSTRAINT NN_DPT_NAME NOT NULL,
		GR_NO CHAR(20) CONSTRAINT NN_GRNUM NOT NULL,
 		EMAIL_ID VARCHAR2(50) CONSTRAINT NN_EML_ID NOT NULL,
 		PAYMENT_STATUS CHAR(7) CONSTRAINT NN_PAY_STATUS NOT NULL,
		ORDER_STATUS CHAR(10) CONSTRAINT NN_ORDER_STATUS NOT NULL,
		ORDER_DATE DATE CONSTRAINT NN_ORDER_DATE NOT NULL,
 		CONSTRAINT FK_USERS_GR_NO FOREIGN KEY(GR_NO) REFERENCES USERS_DETAILS(GR_NO)
 	);
 ****/

/**** FOOD_TABLE	
 	CREATE TABLE FOOD(
 		FOOD_ID NUMBER(7) CONSTRAINT NN_FOOD_ID NOT NULL,
 		FOOD_PRICE NUMBER(6,2) CONSTRAINT NN_FOOD_PRICE NOT NULL,
 		DISCOUNT NUMBER(4,2) CONSTRAINT NN_DISCOUNT_FOOD NOT NULL,
 		FOOD_ITEM VARCHAR2(50) CONSTRAINT NN_FOOD_ITEM NOT NULL,
 		FOOD_ITEM_CATEGORY CHAR(3) CONSTRAINT NN_FOOD_CATGRY NOT NULL,
 		FOOD_IMG VARCHAR2(100) CONSTRAINT NN_FOOD_IMG NOT NULL,
 		FOOD_DESCRIPTION LONG CONSTRAINT NN_FOOD_DESC NOT NULL,
 		CONSTRAINT PK_FD_ID PRIMARY KEY (FOOD_ID)
 	);
 ****/

/**** ORDERS_DETAILS_TABLE	
 	CREATE TABLE ORDERS_DETAILS(
 		ORDER_ID NUMBER(8),
		FOOD_ID NUMBER(7),
 		FOOD_PRICE NUMBER(6,2),
 		DISCOUNT NUMBER(4,2),
 		FOOD_ITEM VARCHAR2(50),
 		FOOD_ITEM_CATEGORY CHAR(3),
 		CONSTRAINT FK_ORDERS_ORDR_ID FOREIGN KEY(ORDER_ID) REFERENCES ORDERS(ORDER_ID) ON DELETE CASCADE,
		CONSTRAINT FK_FD_ID FOREIGN KEY(FOOD_ID) REFERENCES FOOD(FOOD_ID)
 	);	
 ****/

/**** INVENTORY_TABLE
	CREATE TABLE INVENTORY(
		PRODUCT_CODE NUMBER(10) CONSTRAINT PK_PROD_CODE PRIMARY KEY, 
		PRODUCT_NAME VARCHAR2(50) CONSTRAINT NN_PROD_NAME NOT NULL,
		PRODUCT_PRICE NUMBER(6,2) CONSTRAINT NN_PROD_PRICE NOT NULL,
		PRODUCT_QTY NUMBER(6,2) CONSTRAINT NN_PROD_QTY NOT NULL,
		PRODUCT_IMG VARCHAR2(100),
		RETAILER_NAME VARCHAR2(30),
		CONSTRAINT FK_RTL_NUM FOREIGN KEY(RETAILER_NAME) REFERENCES RETAILER_DETAILS(RETAILER_NAME)
	);
